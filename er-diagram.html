ER Diagram — Agricultural Anti-Counterfeit System

AgriSystem Database · SQL Server

erDiagram

    ACCOUNT {
        int      account_id   PK
        varchar  status
        varchar  type
        nvarchar username     UK
        nvarchar password
        nvarchar email        UK
        datetime created_at
    }

    MANAGER {
        int      manager_id        PK
        int      account_id        FK
        nvarchar full_name
        nvarchar managed_id        UK
        nvarchar phone_number
        nvarchar organization_name
    }

    CUSTOMER {
        int      customer_id  PK
        int      account_id   FK
        nvarchar full_name
        nvarchar phone_number
        nvarchar update_log
    }

    STAFF {
        int      staff_id   PK
        int      garden_id  FK
        int      account_id FK
        nvarchar full_name
        nvarchar phone
        nvarchar position
    }

    GARDEN {
        int      garden_id       PK
        int      manager_id      FK
        nvarchar name
        decimal  area
        nvarchar street
        nvarchar city
        nvarchar soil_type
        date     registered_date
    }

    CERTIFICATION {
        int      certification_id PK
        int      garden_id        FK
        nvarchar name
        date     issue_date
        date     expiry_date
        nvarchar issuing_authority
    }

    PRODUCT {
        int      product_id   PK
        int      garden_id    FK
        nvarchar product_name
        nvarchar type
        nvarchar description
        nvarchar quality
        decimal  area
        nvarchar species
        decimal  unit_price
        datetime created_at
    }

    BATCH {
        int      batch_id          PK
        int      product_id        FK
        nvarchar batch_no          UK
        date     harvest_date
        date     expiry_date
        decimal  initial_quantity
        decimal  current_quantity
        varchar  status
        datetime created_at
    }

    TRANSACTION {
        int      transaction_id   PK
        int      customer_id      FK
        datetime transaction_date
        nvarchar street
        nvarchar city
        varchar  status
        decimal  total_amount
        nvarchar notes
    }

    TRANSACTION_DETAIL {
        int     detail_id      PK
        int     transaction_id FK
        int     batch_id       FK
        decimal unit_price
        decimal quantity
        decimal subtotal
    }

    COUNTERFEIT_REPORT {
        int      report_id     PK
        int      customer_id   FK
        int      product_id    FK
        int      resolved_by   FK
        datetime report_date
        nvarchar location
        date     detected_date
        nvarchar fake_method
        nvarchar evidence_url
        varchar  status
        nvarchar conclusion
        datetime resolved_at
    }

    ACCOUNT      ||--o|  MANAGER             : "owns"

    MANAGER      ||--o{  GARDEN              : "manages"
    MANAGER      ||--o{  COUNTERFEIT_REPORT  : "resolves"
    GARDEN       ||--o{  STAFF               : "employs"
    GARDEN       ||--o{  CERTIFICATION       : "has"
    GARDEN       ||--o{  PRODUCT             : "produces"
    PRODUCT      ||--o{  BATCH               : "splits into"
    PRODUCT      ||--o{  COUNTERFEIT_REPORT  : "reported in"
    CUSTOMER     ||--o{  TRANSACTION         : "places"
    CUSTOMER     ||--o{  COUNTERFEIT_REPORT  : "reports"
    TRANSACTION  ||--o{  TRANSACTION_DETAIL  : "contains"
    BATCH        ||--o{  TRANSACTION_DETAIL  : "included in"
